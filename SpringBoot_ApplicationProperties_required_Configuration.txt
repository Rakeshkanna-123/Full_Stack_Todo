-------Configurations required for jdbc through hibernate using jpa -------these are written and stored in resources/application.properties

spring.application.name=todo (default)

//used to change the port from 8080 to 8081 or any other ports
server.port=8081 (optional)

//used for h2 database configuration
spring.datasource.url=jdbc:h2:mem:tododb
spring.datasource.driverClassName=org.h2.Driver
spring.datasource.user=admin
spring.database.password=1234

//used to enable the h2 database console on the website using localhost:8081/h2-console
spring.h2.console.enabled=true

//used to configure jpa 
spring.jpa.database-platform=org.hibernate.dialect.H2Dialect
spring.jpa.hibernate.ddl-auto=update
